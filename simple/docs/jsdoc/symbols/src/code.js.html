 29     /**
 30      * Source setup, values initialize
 31      * @param {string} Befunge Source Codes.

 33     jsBefunge.Code.prototype.init = function(src){
 34         // init values
 35         this.Stack.clear();
 36         this.Result.clear();
 37         this.InputBuffer.pos = 0;
 38         this.direction = 6;
 39         this.pos       = [0,0];
 40         this.c         = "";
 41         this.lenmax    = 0;
 42         this.running   = false;
 43         this.timer     = 0;
 44         this.strings   = false;
 45         
 46         // source setup
 47         this.source = src.split("\n");
 48         for(var i in this.source) {
 49             if(this.lenmax < this.source[i].length)
 50                 this.lenmax = this.source[i].length;
 51         }
 52         for(var i in this.source) {
 53             if(this.lenmax > this.source[i].length)
 54                 this.source[i] += new Array(this.lenmax - this.source[i].length + 1).join(' ');
 55         }
 56 
 57         // first command
 58         this.c = this.source[0].charAt(0);
 59         
 60         // init event
 61         this.eventFire("Code.init", this);
 62     };

 64     /**
 65      * Move next position, command
 66      */
 67     jsBefunge.Code.prototype.next = function() {
 68         if(!this.running) return;
 69 
 70         // moving cursor
 71         this.pos[0] += this.direction == 4 ? -1 : (this.direction == 6 ? 1 : 0);
 72         this.pos[1] += this.direction == 8 ? -1 : (this.direction == 2 ? 1 : 0);
 73         
 74         // start-end loop
 75         if(this.pos[0] > this.lenmax - 1) this.pos[0] = 0;
 76         if(this.pos[0] < 0) this.pos[0] = this.lenmax - 1;
 77         if(this.pos[1] > this.source.length - 1) this.pos[1] = 0;
 78         if(this.pos[1] < 0) this.pos[1] = this.source.length - 1;
 79         
 80         // command update
 81         this.c = this.source[this.pos[1]].charAt(this.pos[0]);
 82     };

 84     /**
 85      * Code run
 86      * @param {string} Befunge Source
 87      * @param {boolean} init only flag ( default: false )
 88      */
 89     jsBefunge.Code.prototype.run = function(src, initOnly) {
 90         var _this = this;
 91         if(_this.running == false) {
 92             _this.stop(true);
 93             _this.init(src);
 94             _this.running = true;
 95         }
 96         if(!initOnly) {
 97             _this.timer = setInterval(function() {
 98                 // step exec
 99                 if(!_this.step()) {
100                     _this.stop();
101                     _this.eventFire("Code.end", _this);
102                 }
103             }, this.wait);
104         }
105             
106         // code run event
107         _this.eventFire("Code.run", this);
108     };

139     /**
140      * Code one step action practical use function
141      * @return {boolean} code stopped:false, continue:true
142      */
143     jsBefunge.Code.prototype._step = function() {
144         if(!this.running) return false; // code running check
145         
146         // string command
147         if(this.strings) {
148             if(this.c != '"') {
149                 this.Stack.push( this.c.charCodeAt(0) );
150                 this.next();
151             } else {
152                 this.next();
153                 this.strings = false;
154             }
155             
156             /*// BreakPoint
157             if($(BefungeElements.RunCode + " tr:eq(" + this.pos[1]  + ") td:eq(" + this.pos[0] + ")").hasClass("break")) {
158                 this.BreakPoint();
159                 return false;
160             }*/
161             return true;
162         }
163         
164         // command exec
165         switch(this.c){
166             // skip next
167             case '#':
168                 this.next();
169                 break;
170                 
171             // direction
172             case 'v':
173                 this.direction = 2;
174                 break;
175             case '<':
176                 this.direction = 4;
177                 break;
178             case '>':
179                 this.direction = 6;
180                 break;
181             case '^':
182                 this.direction = 8;
183                 break;
184             case '_': case '|':
185                 var d = this.Stack.pop();
186                 this.direction = this.c == '_' ? (d == 0 ? 6 : 4) : (d == 0 ? 2 : 8)
187                 break;
188             case '?':
189                 this.direction = parseInt(Math.random() * 4) * 2;
190                 break;
191                 
192             // literal
193             case '0': case '1': case '2':
194             case '3': case '4': case '5':
195             case '6': case '7': case '8':
196             case '9':
197                 this.Stack.push( parseInt(this.c) );
198                 break;
199             
200             case '"':
201                 this.strings = true;
202                 break;
203                 
204                 
205             // user input
206             case '&':
207                 this.Stack.push( String.fromCharCode(this.InputBuffer.get()) );
208                 break;
209             case '~':
210                 this.Stack.push( this.InputBuffer.get() );
211                 break;
212                 
213             // output    
214             case '.':
215                 this.Result.add( this.Stack.pop() + ' ' );
216                 break;
217             case ',':
218                 this.Result.add( String.fromCharCode(this.Stack.pop()) );
219                 break;
220                 
221             // calc
222             case '+':
223             case '-':
224             case '*':
225             case '/':
226             case '%':
227                 var a = this.Stack.pop();
228                 var b = this.Stack.pop();
229                 this.Stack.push( eval("(" + b + ")" + this.c + "(" + a + ")") );
230                 break;
231                 
232             case '`':
233                 var n = this.Stack.pop();
234                 this.Stack.push( this.Stack.pop() > n ? 1 : 0 );
235                 break;
236             case '!':
237                 this.Stack.push( this.Stack.pop() ? 0 : 1 );
238                 break;
239                 
240             // stack
241             case ':':
242                 var n = this.Stack.pop();
243                 this.Stack.push(n);
244                 this.Stack.push(n);
245                 break;
246                 
247             case "\\":
248                 var a = this.Stack.pop();
249                 var b = this.Stack.pop();
250                 this.Stack.push(a);
251                 this.Stack.push(b);
252                 break;
253                 
254             case '$':
255                 this.Stack.pop();
256                 break;
257                     
258             case 'g':
259                 var y = this.Stack.pop();
260                 var x = this.Stack.pop();
261                 var v = this.source[y].charCodeAt(x);
262                 this.Stack.push(v);
263                 break;
264                 
265             case 'p':
266                 var y = this.Stack.pop();
267                 var x = this.Stack.pop();
268                 var v = this.Stack.pop();
269                 this.source[y] =
270                     this.source[y].substring(0, x)
271                     + String.fromCharCode(v)
272                     + (x+1 == this.source[y].length ? '' : this.source[y].substring(x+1, this.source[y].length));
273                 break;
274                 
275             // nop
276             case ' ':
277                 break;
278             
279             // end
280             case '@':
281                 return false;
282                 break;
283                 
284             default:
285                 break;
286         }
287         this.next();
288         return true;
289     };
